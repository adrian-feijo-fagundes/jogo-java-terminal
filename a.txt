function calcularDificuldade(sorteInimigo, velocidadeInimigo, distancia, condicao) {
    return (sorteInimigo * 0.5) + (velocidadeInimigo * 0.5) + distancia + condicao;
}

function calcularEsquiva(sorteJogador, velocidadeJogador, dificuldade) {
    const chanceBase = (sorteJogador * 0.4) + (velocidadeJogador * 0.6) - dificuldade;
    const dado = Math.random() * 100;
    return dado <= chanceBase;
}

// Exemplo de ataque
const sorteInimigo = 40;
const velocidadeInimigo = 60;
const distancia = 10; // Curta distância
const condicao = 10; // Nevoeiro

const dificuldade = calcularDificuldade(sorteInimigo, velocidadeInimigo, distancia, condicao);

// Atributos do jogador
const sorteJogador = 60;
const velocidadeJogador = 80;

if (calcularEsquiva(sorteJogador, velocidadeJogador, dificuldade)) {
    console.log("Esquiva bem-sucedida!");
} else {
    console.log("O ataque acertou!");
